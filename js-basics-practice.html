一、变量定义（var / let / const）

// var —— 函数作用域，允许重复声明，存在变量提升
var city = "北京";

// let —— 块级作用域，不允许重复声明
let score = 99;

// const —— 块级常量，声明后不可重新赋值
const PI = 3.14159;

// 演示块级作用域
{
  let blockVar = "块内变量";
  console.log("块内可访问：", blockVar);
}

二、变量类型（基本类型 & 引用类型）

三、表达式（算术、比较、逻辑、三元、空值合并）

// 算术运算符
5 + 3;   // 8
10 - 4;  // 6
3 * 7;   // 21
15 / 4;  // 3.75
15 % 4;  // 3（取余）
2 ** 8;  // 256（幂运算）

// 比较运算符（== vs ===）
"5" ==  5;  // true  （类型转换后比较）
"5" === 5;  // false （严格相等，类型不同）

// 逻辑运算符
true && false; // false
true || false; // true
!true;         // false

// 三元运算符
let age = 20;
let status = age >= 18 ? "成年人" : "未成年";

// 空值合并运算符 (??)
let nickname = null ?? "匿名用户"; // "匿名用户"

// 可选链 (?.)
let user = { profile: { city: "上海" } };
let city2 = user?.profile?.city;   // "上海"
let zip   = user?.address?.zip;    // undefined（不报错）

四、流程控制（if / switch / for / while / break）

// if / else if / else
function classify(score) {
  if      (score >= 90) return "优秀 🏆";
  else if (score >= 75) return "良好 👍";
  else if (score >= 60) return "及格 ✅";
  else                  return "不及格 ❌";
}

// switch
function dayName(n) {
  switch(n) {
    case 1: return "周一";
    case 2: return "周二";
    case 3: return "周三";
    case 4: return "周四";
    case 5: return "周五";
    case 6: return "周六";
    case 7: return "周日";
    default: return "未知";
  }
}

// for 循环
let sum = 0;
for (let i = 1; i <= 100; i++) { sum += i; }
// sum === 5050

// for...of（遍历可迭代对象）
const fruits = ["🍎苹果", "🍌香蕉", "🍊橙子"];
for (const f of fruits) { console.log(f); }

// while 循环
let n = 1, product = 1;
while (n <= 5) { product *= n; n++; }
// 5! = 120

// break / continue
for (let i = 0; i < 10; i++) {
  if (i === 3) continue;  // 跳过 3
  if (i === 7) break;     // 到 7 停止
  console.log(i);
}

五、函数（声明 / 表达式 / 箭头 / 高阶 / 递归 / 闭包）

// 1. 函数声明（存在提升）
function greet(name = "世界") {
  return `你好，${name}！`;
}

// 2. 函数表达式
const square = function(x) { return x * x; };

// 3. 箭头函数
const cube = x => x ** 3;
const add  = (a, b) => a + b;

// 4. 剩余参数 & 展开运算符
function sumAll(...nums) {
  return nums.reduce((acc, n) => acc + n, 0);
}
sumAll(1, 2, 3, 4, 5); // 15

// 5. 高阶函数（函数作为参数 / 返回值）
const doubled = [1,2,3,4,5].map(x => x * 2);
const evens   = [1,2,3,4,5].filter(x => x % 2 === 0);
const total   = [1,2,3,4,5].reduce((a,b) => a + b, 0);

// 6. 递归
function factorial(n) {
  return n <= 1 ? 1 : n * factorial(n - 1);
}
factorial(6); // 720

// 7. 闭包（Closure）
function makeCounter(start = 0) {
  let count = start;
  return {
    inc:   () => ++count,
    dec:   () => --count,
    reset: () => { count = start; },
    value: () => count
  };
}
const counter = makeCounter(10);

综合演示：学生成绩统计

// 使用函数 + 数组高阶函数综合处理数据
const students = [
  { name: "Alice", scores: [88, 92, 79] },
  { name: "Bob",   scores: [65, 71, 58] },
  { name: "Carol", scores: [95, 98, 100] },
  { name: "Dave",  scores: [45, 55, 40]  },
  { name: "Eve",   scores: [72, 80, 76]  },
];

const avg  = arr => arr.reduce((a,b) => a+b, 0) / arr.length;
const rank = s  => s >= 90 ? "优秀" : s >= 75 ? "良好" : s >= 60 ? "及格" : "不及格";

const report = students
  .map(s => ({ ...s, avg: avg(s.scores).toFixed(1), rank: rank(avg(s.scores)) }))
  .sort((a, b) => b.avg - a.avg);
