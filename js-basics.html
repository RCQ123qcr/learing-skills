🧩 JavaScript 基础知识练习

📦一、变量的定义 
// var —— 函数作用域，可重复声明，存在变量提升
var name = "小明";
var name = "小红";   // 允许重复声明
console.log(name);   // 小红

// let —— 块级作用域，不可重复声明
let age = 18;
age = 20;           // 可以重新赋值
console.log(age);    // 20

// const —— 块级作用域，声明后不可重新赋值
const PI = 3.14159;
console.log(PI);     // 3.14159

// 变量提升演示
console.log(hoisted); // undefined（不报错）
var hoisted = "我被提升了";
▶ 运行输出
🏷️二、变量类型 
// 基本类型（Primitive Types）
let str    = "Hello, JS";        // string
let num    = 42;                  // number
let float  = 3.14;               // number（浮点）
let bool   = true;               // boolean
let n      = null;               // null
let u      = undefined;          // undefined

// 引用类型
let obj    = { city: "北京", pop: 2200 }; // object
let arr    = [1, 2, 3, 4, 5];           // array（object）
let fn     = function() {};              // function

// typeof 检测类型
console.log(typeof str);    // "string"
console.log(typeof num);    // "number"
console.log(typeof bool);   // "boolean"
console.log(typeof n);      // "object"（JS历史遗留bug）
console.log(typeof u);      // "undefined"
console.log(typeof obj);    // "object"
console.log(typeof arr);    // "object"
console.log(typeof fn);     // "function"
console.log(Array.isArray(arr)); // true（判断数组专用）
▶ 运行输出
🔢三、表达式 
// 算术运算符
let a = 10, b = 3;
console.log(a + b);   // 13
console.log(a - b);   // 7
console.log(a * b);   // 30
console.log(a / b);   // 3.333...
console.log(a % b);   // 1（取余）
console.log(a ** b);  // 1000（幂运算）

// 比较运算符
console.log(5 ==  "5");  // true（只比较值）
console.log(5 === "5");  // false（值和类型都比）
console.log(5 !== "5");  // true
console.log(10 > 8);     // true

// 逻辑运算符
console.log(true && false); // false（与）
console.log(true || false); // true（或）
console.log(!true);          // false（非）

// 三元运算符
let score = 75;
let grade = score >= 60 ? "及格" : "不及格";
console.log(grade);  // 及格

// 字符串拼接 与 模板字符串
let user = "小明";
console.log("你好，" + user + "！");       // 传统拼接
console.log(`你好，${user}！分数：${score}`); // 模板字符串
▶ 运行输出
🔀四、流程控制 
// if / else if / else
let s = 82;
if (s >= 90) {
  console.log("优秀");
} else if (s >= 75) {
  console.log("良好");   // ← 执行这里
} else {
  console.log("一般");
}

// switch
let day = 3;
switch (day) {
  case 1: console.log("周一"); break;
  case 2: console.log("周二"); break;
  case 3: console.log("周三"); break; // ← 执行这里
  default: console.log("其他");
}

// for 循环
let sum = 0;
for (let i = 1; i <= 5; i++) {
  sum += i;
}
console.log("1~5 之和：" + sum);  // 15

// while 循环
let n = 1, product = 1;
while (n <= 5) {
  product *= n++;
}
console.log("5! = " + product);  // 120

// for...of 遍历数组
const fruits = ["苹果", "香蕉", "橙子"];
for (const f of fruits) {
  console.log("水果：" + f);
}
▶ 运行输出
⚙️五、函数 
// 1. 函数声明
function greet(name) {
  return `你好，${name}！`;
}
console.log(greet("小明"));  // 你好，小明！

// 2. 函数表达式
const add = function(a, b) {
  return a + b;
};
console.log(add(3, 7));  // 10

// 3. 箭头函数
const multiply = (a, b) => a * b;
console.log(multiply(4, 5));  // 20

// 4. 默认参数
function power(base, exp = 2) {
  return base ** exp;
}
console.log(power(3));     // 9（exp 默认为 2）
console.log(power(2, 10)); // 1024

// 5. 递归函数 —— 斐波那契数列
function fib(n) {
  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
}
console.log("fib(8) = " + fib(8));  // 21

// 6. 闭包 —— 计数器工厂
function makeCounter() {
  let count = 0;
  return () => ++count;
}
const counter = makeCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3
▶ 运行输出
