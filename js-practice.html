一、变量的定义（var / let / const）

// var —— 函数作用域，可重复声明，存在变量提升
var city = "北京";

// let —— 块级作用域，不可重复声明
let score = 95;
score = 98;            // 允许重新赋值

// const —— 块级作用域，声明后不可重新赋值
const PI = 3.14159;
// PI = 3;             // ❌ 报错：常量不可修改

console.log(city, score, PI);
▶ 控制台输出

二、变量类型（7 种基本 + 引用类型）

// 基本类型
let name    = "Alice";           // String
let age     = 20;                // Number
let price   = 9.99;              // Number (浮点)
let isPass  = true;              // Boolean
let nothing = null;              // Null
let undef;                       // Undefined
const id    = Symbol("uid");    // Symbol

// 引用类型
let arr  = [1, 2, 3];            // Array
let obj  = { x: 10, y: 20 };     // Object

// typeof 检测类型
console.log(typeof name, typeof age, typeof isPass,
            typeof nothing, typeof undef, typeof arr);

三、表达式（算术 / 比较 / 逻辑 / 三元）

let a = 10, b = 3;

// 算术表达式
console.log("加:", a+b, "减:", a-b, "乘:", a*b,
            "除:", a/b, "取余:", a%b, "幂:", a**b);

// 比较表达式（== 宽松 vs === 严格）
console.log("10 == '10':", 10 ==  '10');   // true (类型转换)
console.log("10 === '10':", 10 === '10');   // false (严格)

// 逻辑表达式
console.log("&&:", true && false,
            "||:", true || false,
            "!:", !true);

// 三元表达式
let grade = a >= 10 ? "优秀" : "良好";
console.log("成绩等级:", grade);
▶ 控制台输出
🔀

四、流程控制（if / switch / for / while）

// ① if / else if / else
let points = 78;
if (points >= 90)      console.log("A 优秀");
else if (points >= 75) console.log("B 良好");
else if (points >= 60) console.log("C 及格");
else                   console.log("D 不及格");

// ② switch
let day = 3;
switch (day) {
  case 1: console.log("周一"); break;
  case 2: console.log("周二"); break;
  case 3: console.log("周三"); break;
  default: console.log("其他");
}

// ③ for 循环 —— 累加 1~5
let sum = 0;
for (let i = 1; i <= 5; i++) sum += i;
console.log("1+2+3+4+5 =", sum);

// ④ while 循环
let n = 1, product = 1;
while (n <= 5) { product *= n; n++; }
console.log("5! =", product);

// ⑤ for...of 遍历数组
let fruits = ["苹果", "香蕉", "橙子"];
let fruitStr = "";
for (let f of fruits) fruitStr += f + " ";
console.log("水果:", fruitStr.trim());
▶ 控制台输出
⚙️

五、函数（声明式 / 表达式 / 箭头函数 / 递归 / 闭包）

// ① 函数声明（有提升）
function greet(name = "同学") {
  return `你好，${name}！`;
}
console.log(greet("小明"), greet());

// ② 函数表达式
const square = function(x) { return x * x; };
console.log("3² =", square(3));

// ③ 箭头函数（简洁语法）
const add = (x, y) => x + y;
console.log("7 + 8 =", add(7, 8));

// ④ 递归函数 —— 斐波那契数列
function fib(n) {
  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
}
console.log("fib(8) =", fib(8));

// ⑤ 闭包 —— 私有计数器
function makeCounter() {
  let count = 0;
  return () => ++count;
}
const counter = makeCounter();
console.log("计数:", counter(), counter(), counter());
▶ 控制台输出
